## Exercise 5. reduce

    city: 'madrid',
    paymentMethod: {
      'credit-card': '4242424242424242',
    },
  },
};

// add you solution here
const prices = input.orders.map(item => item.price);
function suma (a,b){return a+b};
const total = prices.reduce(suma);
input["totalPrice"] = total;
